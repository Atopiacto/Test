// Memeriksa apakah wallet Solana (misalnya Phantom) sudah terinstal if ("solana" in window) { const provider = window.solana; // Tambahkan event listener untuk tombol "Connect Wallet" document.getElementById("connectButton").addEventListener("click", async () => { try { // Minta koneksi ke wallet const response = await provider.connect(); // Tampilkan alamat publik pengguna console.log("Wallet terhubung:", response.publicKey.toString()); alert("Wallet terhubung! Alamat Anda: " + response.publicKey.toString()); } catch (err) { // Tangani error jika pengguna menolak koneksi console.error("Koneksi ditolak:", err); alert("Koneksi ditolak."); } }); } else { // Beri tahu pengguna untuk menginstal wallet Solana alert("Silakan instal ekstensi wallet Solana, seperti Phantom!"); }
